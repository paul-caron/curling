117inline void trim(std::string& s) {
118 s.erase(s.begin(), std::find_if(s.begin(), s.end(), [](unsigned char ch){
119 return !std::isspace(ch);
120 }));
121 s.erase(std::find_if(s.rbegin(), s.rend(), [](unsigned char ch){
122 return !std::isspace(ch);
123 }).base(), s.end());
124}

138inline size_t HeaderCallback(char* buffer, size_t size, size_t nitems, void* userdata) {
139 auto* headerMap = static_cast<std::map<std::string, std::vector<std::string>>*>(userdata);
140 std::string headerLine(buffer, size * nitems);
141
142 if (headerLine.empty()) return 0; // skip the separation line
143
144 auto colonPos = headerLine.find(":");
145 if (colonPos != std::string::npos) {
146 std::string key = headerLine.substr(0, colonPos);
147 std::string value = headerLine.substr(colonPos + 1);
148 detail::trim(key);
149 detail::trim(value);
150 detail::toLowerCase(key);
151 (*headerMap)[key].push_back(value);
152 }
153
154 return size * nitems;
155}

227struct Response {
228 long httpCode;
229 std::string body;
230 std::map<std::string, std::vector<std::string>> headers;
231
232 std::string toString() const {
233 std::ostringstream oss;
234 oss << "status: " << httpCode << "\nbody:\n" << body << "\nheaders:\n";
235 for (auto const& h : headers) {
236 oss << h.first << ": ";
237 for (auto const& v : h.second) oss << v << " ";
238 oss << "\n";
239 }
240 return oss.str();
241 }
242 std::vector<std::string> getHeader(const std::string& key) const {
243 std::string lowered = key;
244 detail::toLowerCase(lowered);
245 auto it = headers.find(lowered);
246 return (it != headers.end()) ? it->second : std::vector<std::string>{};
247 }

526inline int ProgressCallbackBridge(void* clientp, curl_off_t dltotal, curl_off_t dlnow,
527 curl_off_t ultotal, curl_off_t ulnow) {
528 auto* req = static_cast<Request*>(clientp);
529 if (req->progressCallback) {
530 bool shouldCancel = req->progressCallback(dltotal, dlnow, ultotal, ulnow);
531 return shouldCancel ? 1 : 0; // Returning non-zero aborts transfer
532 }
533 return 0;
