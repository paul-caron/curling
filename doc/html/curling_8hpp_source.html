93inline void trim(std::string& s) {
94 s.erase(s.begin(), std::find_if(s.begin(), s.end(), [](unsigned char ch){
95 return !std::isspace(ch);
96 }));
97 s.erase(std::find_if(s.rbegin(), s.rend(), [](unsigned char ch){

177struct Response {
178 long httpCode;
179 std::string body;
180 std::map<std::string, std::vector<std::string>> headers;
181
182 std::string toString() const {
183 std::ostringstream oss;
184 oss << "status: " << httpCode << "\nbody:\n" << body << "\nheaders:\n";
185 for (auto const& h : headers) {
186 oss << h.first << ": ";
187 for (auto const& v : h.second) oss << v << " ";
188 oss << "\n";
189 }
190 return oss.str();
191 }
192 std::vector<std::string> getHeader(const std::string& key) const {
193 std::string lowered = key;
194 detail::toLowerCase(lowered);
195 auto it = headers.find(lowered);
196 return (it != headers.end()) ? it->second : std::vector<std::string>{};
197 }
198};
